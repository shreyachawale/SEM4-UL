COST PRICE
#1/bin/bash

echo "Enter cost price: "
read cp 
echo "Enter selling price: "
read sp

if [[ sp -gt cp ]]; then
    profit=$(expr $sp-$cp)
    echo "Profit of $(( profit ))"
elif [[ cp -gt sp ]]; then
    loss=$(expr $cp-$sp)
    echo "Loss of $(( loss ))"
else
    echo "Both are equal"
fi

CALCULATOR
